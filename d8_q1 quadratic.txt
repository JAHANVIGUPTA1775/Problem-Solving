#include <iostream>
#include<cmath>

using namespace std;

int main()
{
    float a,b,c,d,n;
    float r1,r2;
    cin>>a>>b>>c;
    d=(b*b)-(4*a*c);
    if(d==0){
        cout<<"roots are equal"<<endl;
        r1= -b/(2*a);
        //r1= -b/(2*a);
        cout<<r1<<endl;
    }
    
    if(d>0){
        r1= (-b+sqrt(d))/2*a;
        r2= (-b-sqrt(d))/2*a;
        cout<<r1<<" "<<r2<<endl;
    }
    if(d<0){
        cout<<"roots are imaginary"<<endl;
    }

    return 0;
}
